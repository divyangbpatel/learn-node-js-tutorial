project to use node js
    npm init
    npm --version
    node --version
    npm i
    node file_name only 

install package
    npm i -g nodemon(pkg_name)
    -g means global

dev dependincy to only use not global
    npm i --save-dev nodemon(pkg_name)

install express
    npm install express --save(optional)

npm uninstall
    npm uninstall nodemon(pkg_name)


install angular cli
    npm i @angular/cli
